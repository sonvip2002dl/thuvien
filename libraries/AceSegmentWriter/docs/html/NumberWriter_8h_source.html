291  uint8_t writeHexCharsInsideBoxAt(
292  uint8_t pos,
293  const hexchar_t s[],
294  uint8_t len,
295  int8_t boxSize) {
296 
297  uint8_t absBoxSize = (boxSize < 0) ? -boxSize : boxSize;
298 
299  // if the box is too small, print normally
300  if (len >= absBoxSize) {
301  writeInternalHexCharsAt(pos, s, len);
302  return len;
303  }
304 
305  // Print either left justified or right justified inside box
306  uint8_t padSize = absBoxSize - len;
307  if (boxSize < 0) {
308  // left justified
309  writeInternalHexCharsAt(pos, s, len);
310  pos += len;
311  while (padSize--) writeInternalHexCharAt(pos++, kHexCharSpace);
312  } else {
313  // right justified
314  while (padSize--) writeInternalHexCharAt(pos++, kHexCharSpace);
315  writeInternalHexCharsAt(pos, s, len);
316  }
317 
318  return absBoxSize;
319  }

334  static uint8_t toDecimal(uint16_t num, hexchar_t buf[], uint8_t bufSize) {
335  uint8_t pos = bufSize;
336  while (true) {
337  if (num < 10) {
338  buf[--pos] = num;
339  break;
340  }
341  uint16_t quot = num / 10;
342  buf[--pos] = num - quot * 10;
343  num = quot;
344  }
345  return pos;
346  }
